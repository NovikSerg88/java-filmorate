<details><summary>Примеры запросов:</summary>

1. _Получение списка всех пользователей:_
SELECT * "/n"
FROM users /n
2. _Получение пользователя по id = 1:_
SELECT *
FROM users
WHERE user_id=1;
3. _Получение списка всех фильмов:_
SELECT *
FROM films;
4. _Получение фильма по id =1:_
SELECT *
FROM films
WHERE film_id=1;
